
import requests # usado para requisições HTTPS (baixar dados de uma URL)

URL = "https://ddragon.leagueoflegends.com/cdn/14.1.1/data/pt_BR/champion.json" 
# API ESTÁTICA | PÚBLICA != API DINÂMICA


resp = requests.get(URL) #baixa o arquivo JSON!
resp.raise_for_status() # verificação de erros!
data = resp.json() # transforma o texto JSON recebido da API em um \
                   # objeto Python (dicionário)


champions = data["data"]  # chave data = infos dos campeões

for champ_key, champ_info in champions.items(): # retorna (chave, valor)
    nome = champ_info["name"]
    titulo = champ_info.get("title", "")
    tags = champ_info.get("tags", []) 

    print(f"{nome} — {titulo} — {tags}")
